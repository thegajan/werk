ARRAY FORMAT:
    [{"last_updated":"yyyy/mm/dd hh:mm:ss (24 hour time)", "creator_id": id,{task 1},{task 2}...}]
KEYS:
    taskName
    description
    startDate - mm/dd/yyyy hh:mm:ss A
    endDate - "
    status - create or delete
    id - the old id of the task
